        public Student findStudentByEmail(String email) {
            try (Session session = factory.openSession()) {
                String hqlQuery = "FROM Student s WHERE s.email = :email";
                System.out.println("Querying for student with email: " + email);
    
                TypedQuery query = session.createQuery(hqlQuery, Student.class);
                query.setParameter("email", email);
    
                try {
                    return query.getSingleResult();
                } catch (Exception e) {
                    System.out.println("Student not found with email: " + email);
                    return null;
                }
            }
        }
